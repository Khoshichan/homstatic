### Render rows.
function template_preprocess_views_job_board(&$vars) {
  $result = $vars['result'] = $vars['rows'];

  $view = &$vars['view'];
  $options  = $view->style_plugin->options;
  $handler  = $view->style_plugin;
  $renders = $handler->render_fields($result);
}
