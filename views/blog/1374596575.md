#### Get menu wildcard from path.
    /**
     * Get menu wildcard from path.
     *
     * @param $path
     *   The path, for example node/5.
     *
     * @return
     *   Returns wildcard.
     */
    function get_menu_wildcard($path) {
      $item = menu_get_item($path);
      return $item['path'];
    }
